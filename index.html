Classtrophobic ES5

Zero Runtime, Babel Proof, Classes.

const Class = require('classtrophobic-es5');

const List = Class({
  extends: Array,
  static: {
    import: function (o) {
      return List.from(o);
    }
  },
  constructor: function () {
    var self = this.super();
    self.push.apply(self, arguments);
    return self;
  },
  push() {
    this.super.push.apply(this, arguments);
    return this;
  }
});
